145void buildAligment(DirectionMat *mat, const string &seq1, const string &seq2, array<string, 3> &output)
146{
147 size_t i = mat->getRows() - 1;
148 size_t j = mat->getCols() - 1;
149
150 while (i >= 1 || j >= 1)
151 {
152 uint8_t val = mat->at(i, j);
153
154 switch (val)
155 {
156 case VERTICAL:
157 output[0].push_back(seq1.at(i - 1));
158 output[1].push_back(' ');
159 output[2].push_back('-');
160 i--;
161 break;
162
163 case DIAGONAL:
164 output[0].push_back(seq1.at(i - 1));
165 output[1].push_back((seq1.at(i - 1) == seq2.at(j - 1)) ? '|' : '*');
166 output[2].push_back(seq2.at(j - 1));
167 i--;
168 j--;
169 break;
170
171 case HORIZONTAL:
172 output[0].push_back('-');
173 output[1].push_back(' ');
174 output[2].push_back(seq2.at(j - 1));
175 j--;
176 break;
177
178 default:
179 break;
180 }
181 }
182
183 reverse(output[0].begin(), output[0].end());
184 reverse(output[1].begin(), output[1].end());
185 reverse(output[2].begin(), output[2].end());
186}
